Introduction à RStudio

RStudio est un environnement de développement intégré (IDE) pour le langage R. Voici tout ce qu'il faut savoir pour démarrer.

📌 Qu'est-ce que R et RStudio ?
  • R : langage de programmation spécialisé en statistiques et analyse de données
  • RStudio : interface graphique qui facilite l'utilisation de R
  • R est le moteur, RStudio est le tableau de bord
  • Les deux sont gratuits et open-source
🔧 Les 4 panneaux de RStudio
  • Script (haut gauche) : où on écrit le code
  • Console (bas gauche) : où R exécute le code
  • Environnement (haut droite) : variables et données
  • Fichiers / Plots / Help (bas droite)

RTypes de données — exemples pratiques
# --- Numeric & Integer ---
population <- 27478249       # numeric
annee      <- 2024L           # integer
class(population)             # "numeric"
class(annee)                  # "integer"

# --- Character ---
region <- "Abidjan"
nchar(region)                 # 7 (nombre de caractères)
toupper(region)               # "ABIDJAN"
tolower(region)               # "abidjan"

# --- Logical ---
vaccinated <- TRUE

sexe <- factor(c("M","F","M","F"))

table(sexe)                   # F:2  M:2

# --- Dates ---
d1 <- as.Date("2024-01-15")

d2 - d1                       # différence en jours

# --- Conversion de types ---
as.numeric("3.14")           # 3.14
as.character(42)              # "42"
as.logical(1)                 # TRUE

Importation des données

R peut lire presque tous les formats de fichiers. Voici les méthodes pour les formats les plus courants.

RLire des fichiers CSV
library(readr)

# CSV standard (séparateur virgule)
df <- read_csv("donnees.csv")

# CSV européen (séparateur point-virgule, décimal virgule)
df <- read_csv2("donnees.csv")

# Préciser l'encodage (important pour les accents)
df <- read_csv("donnees.csv", locale = locale(encoding = "UTF-8"))

# Sauter des lignes d'en-tête
df <- read_csv("donnees.csv", skip = 2)

# Préciser les types de colonnes
df <- read_csv("donnees.csv", col_types = cols(
  district = col_character(),
  annee    = col_integer(),
  valeur   = col_double()
))

Nettoyage des données

Le nettoyage représente souvent 70% du travail d'analyse. Voici les opérations essentielles avec dplyr, tidyr et janitor.

Rdplyr — manipulation et nettoyage
library(dplyr)
library(janitor)

# Nettoyer les noms de colonnes (espaces, accents, majuscules)
df <- df |> clean_names()       # "Nom Région" → "nom_region"

# Sélectionner des colonnes
df2 <- df |> select(district, annee, valeur)
df2 <- df |> select(-colonne_inutile)   # enlever une colonne

# Renommer des colonnes
df <- df |> rename(district = nom_district, taux = couverture_pct)

# Filtrer des lignes
df |> filter(annee == 2023)
df |> filter(region == "Abidjan" & taux > 80)
df |> filter(!is.na(valeur))           # exclure les NA

# Créer / modifier des colonnes
df <- df |> mutate(
  taux_pct  = vaccines / population * 100,
  annee_cat = if_else(annee >= 2020, "récent", "ancien")
)

# Trier
df |> arrange(desc(taux_pct))           # décroissant
df |> arrange(region, annee)             # multi-colonne

# Agréger / résumer
df |>
  group_by(region) |>
  summarise(
    moy_taux  = mean(taux, na.rm = TRUE),
    n_dist    = n(),
    max_taux  = max(taux, na.rm = TRUE)
  )
RValeurs manquantes (NA)
# Détecter les NA
is.na(df$valeur)               # vecteur logique
sum(is.na(df$valeur))          # nombre de NA
colSums(is.na(df))             # NA par colonne

# Supprimer les lignes avec NA
df_propre <- na.omit(df)
df_propre <- df |> drop_na()   # tidyr

# Remplacer les NA
df <- df |> mutate(
  valeur = replace_na(valeur, 0),          # remplacer par 0
  valeur = if_else(is.na(valeur),
                   mean(valeur, na.rm=TRUE), # remplacer par moy
                   valeur)
)

# Supprimer les doublons
df <- df |> distinct()          # toutes colonnes
df <- df |> distinct(id, .keep_all=TRUE)
RManipulation de texte avec stringr
library(stringr)

df <- df |> mutate(
  district = str_trim(district),          # enlever espaces

  district = str_replace_all(             # remplacer
               district, "-", " "),

# Détecter un motif
df |> filter(str_detect(district, "Abidjan"))

# Manipulation de dates
library(lubridate)
df <- df |> mutate(
  date     = dmy(date_texte),              # "15/01/2024"
  annee    = year(date),
  mois     = month(date, label=TRUE),
  trimestre= quarter(date)
)

Fonctions mathématiques et statistiques

R est un langage statistique. Voici les fonctions les plus utiles pour analyser des données de santé ou toute donnée quantitative.

RFonctions de base
x <- c(12, 45, 7, 89, 23, 56)

sum(x)        # 232
mean(x)       # 38.67
median(x)     # 34
min(x)        # 7
max(x)        # 89
range(x)      # 7 89
var(x)        # variance
sd(x)         # écart-type
abs(x)        # valeur absolue
sqrt(x)       # racine carrée
log(x)        # log naturel
log10(x)      # log base 10
exp(x)        # exponentielle
round(x, 2)   # arrondi 2 décimales
ceiling(x)    # arrondi supérieur
floor(x)      # arrondi inférieur
RStatistiques descriptives
# Quantiles / percentiles
quantile(x, 0.25)            # Q1
quantile(x, c(0.25,0.5,0.75)) # Q1 Q2 Q3
IQR(x)                       # interquartile

# Résumé complet
summary(x)

# Corrélation
cor(x, y)                    # Pearson
cor(x, y, method="spearman") # Spearman

# Tests statistiques
t.test(x, y)                 # test T Student
chisq.test(table(df$sexe,     # test Chi2
                 df$vaccin))
wilcox.test(x, y)            # Mann-Whitney

# Régression linéaire
modele <- lm(taux ~ annee + population, data=df)
summary(modele)
coef(modele)
fitted(modele)                # valeurs prédites
RAgrégations avec dplyr (cas pratique santé)
library(dplyr)

# Taux de couverture vaccinale par région et année
synthese <- df |>
  group_by(region, annee) |>
  summarise(
    total_pop     = sum(population, na.rm = TRUE),
    total_vacc    = sum(vaccines,   na.rm = TRUE),
    taux_cov      = round(total_vacc / total_pop * 100, 1),
    nb_districts  = n_distinct(district),
    .groups       = "drop"
  ) |>
  arrange(desc(taux_cov))

# Pourcentages cumulés
df <- df |> mutate(
  cumul = cumsum(valeur),
  rang  = rank(desc(valeur)),
  pct   = valeur / sum(valeur) * 100
)

Graphiques avec ggplot2

ggplot2 est le standard de la visualisation en R. La logique est toujours la même : données + esthétiques + géométries.

🧱 La grammaire de ggplot2
R
ggplot(data = df,            # 1. les données
       aes(x = district,          # 2. les esthétiques (x, y, color...)
           y = taux,
           fill = region)) +
  geom_col() +               # 3. la géométrie (barres)
  labs(title  = "Titre",       # 4. étiquettes
       x      = "District",
       y      = "Taux (%)") +
  theme_minimal()             # 5. thème visuel
RTypes de graphiques essentiels
library(ggplot2)

# ---- HISTOGRAMME ----
ggplot(df, aes(x = age)) +
  geom_histogram(bins = 20, fill = "#2986cc", color = "white") +
  labs(title = "Distribution des âges")

# ---- BARRES ----
ggplot(df, aes(x = region, y = taux, fill = region)) +
  geom_col() +
  coord_flip() +               # barres horizontales
  theme_minimal() +
  theme(legend.position = "none")

# ---- COURBE TEMPORELLE ----
ggplot(df, aes(x = annee, y = taux, color = district, group = district)) +
  geom_line(linewidth = 1.2) +
  geom_point(size = 3) +
  labs(title = "Évolution du taux de couverture")

# ---- NUAGE DE POINTS ----
ggplot(df, aes(x = population, y = taux, color = region)) +
  geom_point(size = 3, alpha = 0.7) +
  geom_smooth(method = "lm", se = TRUE) +   # droite régression
  scale_x_log10()              # axe logarithmique

# ---- BOÎTE À MOUSTACHES ----
ggplot(df, aes(x = region, y = taux, fill = region)) +
  geom_boxplot() +
  geom_jitter(alpha = 0.3, width = 0.2)  # points individuels

# ---- FACETTES (petits multiples) ----
ggplot(df, aes(x = annee, y = taux)) +
  geom_line() +
  facet_wrap(~region, ncol = 3)  # un graphique par région
